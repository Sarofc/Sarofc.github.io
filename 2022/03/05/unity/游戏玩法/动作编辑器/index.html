预览

抽象

配置和逻辑完全分离，配置使用ScriptableObject存储，可替换成其他数据源。

Runtime

  • MotionClip。ScriptableObject，纯数据，存储各种MotionNotify
  • MotionNotify。Motion事件配置，纯数据，实现接口创建MotionNotifySpec
  • MotionNotifySpec。Motion事件逻辑实例，各种逻辑就写在这里，提供Begin、Tick、End事件可供子类重写
  • Motion。运行时实例，引用MotionClip，管理MotionNotifySpec
  • MotionManager。Motion管理器，管理所有Motion实例

Editor，基于unity timeline

  • MotionNotifyTrack。timeline三件套之一，将MotionNotify分轨道排列
  • MotionNotifyAsset。timeline三件套之一，引用MotionNotify，可编辑面板编辑
  • MotionNotifyBehaviour。timeline三件套之一，提供Motion编辑时预览
  • MotionDesigner。
  • MotionEditorController。

using System;
using Newtonsoft.Json;
using Sirenix.OdinInspector;
using UnityEngine;

namespace Saro.Gameplay.Motion
{
    [Serializable]
    public sealed partial class NS_PlayAnimator : MotionNotify
    {
        [SerializeField]
        [ReadOnly]
        private int m_AnimHash;

        [SerializeField]
        private int m_Layer;

        public int AnimHash
        {
            get => m_AnimHash;
            set => m_AnimHash = value;
        }

        public int Layer
        {
            get => m_Layer;
            set => m_Layer = value;
        }

        public override MotionNotifySpec CreateSpec()
        {
            var spec = SharedPool.Rent<NS_PlayAnimatorSpec>();
            spec.BaseConfig = this;
            return spec;
        }
    }

    public sealed class NS_PlayAnimatorSpec : MotionNotifySpec
    {
        public NS_PlayAnimator Config => BaseConfig as NS_PlayAnimator;
        public int AnimHash => Config.AnimHash;
        public int Layer => Config.Layer;

        private Animator m_Animator;
        private float m_OriginalSpeed;

        protected override void NotifyBegin_Internal(IMotionSource source, Motion motion)
        {
            if (!m_Animator)
            {
                m_Animator = source.Animator;
            }
            m_OriginalSpeed = m_Animator.speed;
            m_Animator.speed = motion.PlaybackSpeed;

            if (motion.CrossFade > 0f)
                m_Animator.CrossFadeInFixedTime(AnimHash, motion.CrossFade, Layer);
            else
                m_Animator.Play(AnimHash, Layer);
        }

        protected override void NotifyEnd_Internal(IMotionSource source, Motion motion)
        {
            m_Animator.speed = m_OriginalSpeed;
        }

        protected override void IReferenceClear()
        {
            base.IReferenceClear();

            m_Animator = null;
        }
    }

#if UNITY_EDITOR
    public partial class NS_PlayAnimator
    {
        [JsonIgnore]
        [OnValueChanged("SetAnimHash")]
        public AnimationClip m_AnimClip;

        public override Color GetClipColor()
        {
            return Color.blue;
        }

        public void SetAnimHash()
        {
            if (m_AnimClip)
                m_AnimHash = Animator.StringToHash(m_AnimClip.name);
            else
                m_AnimHash = 0;
        }

        [Button("SetRange")]
        public void SetStartEndFrameUseClipLength()
        {
            if (m_AnimClip)
            {
                m_Range.End = m_Range.Start + m_AnimClip.GetFrameLength();
            }
        }

        protected override void NotifyTick_Preview(GameObject gameObj, double time, int frame)
        {
            base.NotifyTick_Preview(gameObj, time, frame);

            //var animator = gameObj.GetComponent<Animator>();
            m_AnimClip.SampleAnimation(gameObj, (float)time);
        }
    }
#endif
}
